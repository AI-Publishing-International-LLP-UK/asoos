Installation Instructions

# Option 1: Install globally from the downloaded package
npm install -g ./aixtiv-cli-latest.tgz

# Option 2: Install from this URL
npm install -g https://api-for-warp-drive.web.app/aixtiv-cli-latest.tgz

Usage

# Authentication
aixtiv auth:verify --email someone@example.com

# Grant Access
aixtiv agent:grant --email user@example.com --agent agent007 --resource secret-doc --type full

# Revoke Access
aixtiv agent:revoke --email user@example.com --agent agent007 --resource secret-doc

# Scan Resources
aixtiv resource:scan --agent agent007
